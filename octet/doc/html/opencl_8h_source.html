386  /// queue a call
387  cl_event call(size_t num_work_items, size_t work_group_size, cl_event event=0, bool want_res_event=false) {
388  cl_event res_event;
389  work_group_size = work_group_size ? work_group_size : prefered_workgroup_size;
390  opencl *cl = get_cl();
391  if (cl->error_code) return 0;
392 
393  size_t offset = 0;
394  //printf("%08x work items\n", num_work_items);
395  while (num_work_items) {
396  size_t cur_work_items = num_work_items > max_work_items ? max_work_items : num_work_items;
397  if (cur_work_items < work_group_size) {
398  work_group_size = 1;
399  } else if (cur_work_items % work_group_size) {
400  cur_work_items = cur_work_items / work_group_size * work_group_size;
401  }
402  //printf(" %08x %08x %08x\n", offset, cur_work_items, work_group_size);
403 
404  size_t global_work_offset[1] = {offset};
405  size_t global_work_size[1] = {cur_work_items};
406  size_t local_work_size[1] = {work_group_size};
407  cl->error_code = clEnqueueNDRangeKernel(
408  cl->queue, get_obj(), 1, global_work_offset, global_work_size, local_work_size,
409  event != 0, &event, want_res_event ? &res_event : NULL
410  );
411  if (cl->error_code) {
412  log("call: error %s\n", cl->get_cl_error_name(cl->error_code));
413  }
414  cl->flush();
415 
416  offset += cur_work_items;
417  num_work_items -= cur_work_items;
418  }
419  num_args = 0;
420  return res_event;
421  }
